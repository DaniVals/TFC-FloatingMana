TFC_FloatingMana

Tablas:

	- User:
		- idUser - integer(6) - PK
		- email - varchar(50) UNIQUE NOT NULL
		- password - varchar(255) NOT NULL
		- username - varchar(20) NOT NULL
		- profilePic - varchar(150) - UNIQUE (hace ref a un archivo jpg/png)
		- isAuth - integer(1)
	
	- TokenAuth:
		- token - varchar(255) - PK
		- idUser - integer(6) - FK (User, idUser)
		- expirationDate - datetime
	
	- Deck:
		- idDeck - integer(7) PK
		- deckName - varchar(50) NOT NULL
		- cardList - varchar(150) UNIQUE NOT NULL (dirección donde se guardará el archivo json en el servidor)
		- deckOwner - varchar(20) - FK (Profile, userName)
		- type - varchar(100) NOT NULL
	
	- Collection: 
		- idCollection - integer(6) PRIMARY KEY	
		- deckOwner - varchar(20) - FK (Profile, userName)
		- collCardList - varchar(255) NOT NULL (dir donde está el json)
		- collectionValue - decimal(6,2) NOT NULL
		- cardAmount - integer(4) NOT NULL
